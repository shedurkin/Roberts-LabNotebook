library(tidyverse)

files <- c(
  Apul = "https://raw.githubusercontent.com/urol-e5/timeseries_molecular/refs/heads/main/M-multi-species/output/26.6-ElasticNet-stability-selection-2/Apul/Apul_stabsel_results_nonzero.csv",
  Peve = "https://raw.githubusercontent.com/urol-e5/timeseries_molecular/refs/heads/main/M-multi-species/output/26.6-ElasticNet-stability-selection-2/Peve/Peve_stabsel_results_nonzero.csv",
  Pmea = "https://raw.githubusercontent.com/urol-e5/timeseries_molecular/refs/heads/main/M-multi-species/output/26.6-ElasticNet-stability-selection-2/Ptuh/Ptuh_stabsel_results_nonzero.csv"
)

df_list <- imap(files, ~ read_csv(.x) %>% mutate(Species = .y))
df_all  <- bind_rows(df_list)

shared_genes <- Reduce(intersect, map(df_list, ~ unique(.x$Gene)))
cat("Shared genes:", length(shared_genes), "\n")

df_all$Feature_Type <- ifelse(grepl("^lncRNA", df_all$Predictor), "lncRNA",
                   ifelse(grepl("^Cluster", df_all$Predictor), "miRNA", 
                          "GbM"))

df_shared <- df_all %>%
  filter(Gene %in% shared_genes)

table(df_shared$Feature_Type)

Apul — All predictors

# Filter to species
sp_df <- df_shared %>% filter(Species == "Apul")

# Summarise |coef| per gene x feature type
bar_df <- sp_df %>%
  group_by(Gene, Feature_Type) %>%
  summarise(Abs_Coef = sum(Mean_Abs_Coef), .groups = "drop")

# Count distinct predictors per gene
npred_df <- sp_df %>%
  group_by(Gene) %>%
  summarise(N_Predictors = n_distinct(Predictor))

# Order genes by total magnitude
gene_order <- bar_df %>%
  group_by(Gene) %>%
  summarise(Total = sum(Abs_Coef)) %>%
  arrange(desc(Total)) %>%
  pull(Gene)

bar_df$Gene   <- factor(bar_df$Gene,   levels = gene_order)
npred_df$Gene <- factor(npred_df$Gene, levels = gene_order)

# Scaling factor to map N_Predictors onto the coef axis
coef_max  <- max(bar_df %>% group_by(Gene) %>% summarise(T = sum(Abs_Coef)) %>% pull(T))
npred_max <- max(npred_df$N_Predictors)
scale_factor <- coef_max / npred_max

ggplot(bar_df, aes(x = Gene, y = Abs_Coef, fill = Feature_Type)) +
  geom_bar(stat = "identity") +
  geom_point(data = npred_df,
             aes(x = Gene, y = N_Predictors * scale_factor),
             shape = 23, size = 2, fill = "black",
             inherit.aes = FALSE) +
  scale_y_continuous(
    name = "Sum of |Coefficient|",
    sec.axis = sec_axis(~ . / scale_factor, name = "N Predictors")
  ) +
  labs(title = "Apul — all predictors", x = "Gene", fill = "Feature Type") +
  theme_minimal(base_size = 13) +
  theme(axis.text.x = element_text(angle = 45, hjust = 1, size = 6),
        legend.position = "bottom")

barplots

barplot_function <- function(df, species_prefix, stable){
  sp_df <- df %>% filter(Species == species_prefix, Stable == stable)
  
  bar_df <- sp_df %>%
    mutate(Direction = if_else(Mean_Coef >= 0, "pos", "neg")) %>%
    group_by(Gene, Feature_Type, Direction) %>%
    summarise(Coef_Sum = sum(Mean_Coef), .groups = "drop")
  
  gene_order <- bar_df %>%
    group_by(Gene) %>%
    summarise(Net = sum(Coef_Sum)) %>%
    arrange(desc(Gene)) %>%
    pull(Gene)
  
  bar_df$Gene <- factor(bar_df$Gene, levels = gene_order)
  
  feature_colors <- c(
    "miRNA"  = "#E69F00",
    "lncRNA" = "#377EB8",
    "GbM"    = "#009E73"
  )
  
  ggplot(bar_df, aes(x = Gene, y = Coef_Sum, fill = Feature_Type)) +
    geom_bar(stat = "identity") +
    geom_hline(yintercept = 0, linewidth = 0.5) +
    scale_fill_manual(values = feature_colors) +
    labs(title = paste0(species_prefix, 
                        ifelse(stable, " — stable predictors only", " — all predictors")),
         x = "Gene", y = "Sum of Coefficients", fill = "Feature Type") +
    theme_minimal(base_size = 13) +
    theme(axis.text.x = element_text(angle = 45, hjust = 1, size = 6),
          legend.position = "bottom") +
    ylim(c(-1, 1.6))
}

barplot_function(df_shared, "Apul", TRUE)
barplot_function(df_shared, "Peve", TRUE)
barplot_function(df_shared, "Pmea", TRUE)

barplot_function(df_all, "Apul", TRUE)
barplot_function(df_all, "Peve", TRUE)
barplot_function(df_all, "Pmea", TRUE)
